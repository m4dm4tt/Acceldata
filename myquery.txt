SELECT 
  job_id, 
  AVG(salary) 
FROM 
  employees e FULL OUTER JOIN dept d 
WHERE 
  e.dept_id = d.dept_id 
  AND 1 = 1 
  AND e.salary BETWEEN (10000 + 1) AND 100000 * 10 
  AND e.bonus BETWEEN 100 + 2 AND (1000 * 15) 
GROUP BY 
  job_id 
HAVING 
  AVG(salary)< (
    SELECT 
      MAX(
        AVG(j.min_salary)
      ) 
    FROM 
      (
        select 
          * 
        from 
          jobs
      ) as j 
    WHERE 
      j.job_id IN (
        SELECT 
          jh.job_id 
        FROM 
          job_history jh, 
          (
            select 
              * 
            from 
              job_history_archive
          ) jha 
        WHERE jh.hist_id = jha.hist_id 
          AND jh.department_id BETWEEN 50 AND 100 
        ORDER BY 
          jh.job_id
      ) 
      AND j.job_id IN (
        SELECT 
          global_job_type 
        FROM 
          params
      ) 
    GROUP BY 
      j.job_id
  ) 
ORDER BY 
  d.dept_id 
LIMIT 
  10;
